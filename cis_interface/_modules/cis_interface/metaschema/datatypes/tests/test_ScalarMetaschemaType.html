        iattr_imp['_prec'] = 64

            iattr_imp['_prec'] = max_len * 8
    iattr_exp = copy.deepcopy(iattr_imp)
    iattr_exp['_cls'] = '%sMetaschemaType' % t.title()

                   (TestScalarMetaschemaType, ), iattr_exp)
